    var arr = [];
    function randomArr(arr, i) {
        var num = Math.random() * 30 + 2;
        if(i === -1) return;
        if(arr.indexOf(num) > -1) {
            return randomArr(arr, i);
        } else {
            arr[i] = num;
            return randomArr(arr, --i);
        }
    };
